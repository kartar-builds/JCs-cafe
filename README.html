// Café JC's — Google Apps Script
// Apni sheet ID aur email yahan set karo:

const SHEET_NAME = 'Sheet1';
const OWNER_EMAIL = 'kartarsaini305@gmail.com'; // ← Yahan apni email
const CAFE_NAME   = 'Café JC\'s';

function doGet(e) {
  const action = e.parameter.action;
  
  if (action === 'get') return getBookings();
  if (action === 'updateStatus') return updateStatus(e.parameter.id, e.parameter.status);
  
  return ContentService.createTextOutput('OK');
}

function doPost(e) {
  try {
    const data = JSON.parse(e.postData.contents);
    const sheet = SpreadsheetApp.getActiveSpreadsheet().getSheetByName(SHEET_NAME);
    const id    = sheet.getLastRow();
    const time  = new Date().toLocaleString('en-IN');
    
    sheet.appendRow([
      id, time,
      data.fname, data.lname, data.email, data.phone,
      data.date, data.time, data.guests, data.seating,
      data.notes, 'pending'
    ]);
    
    // Owner ko email
    sendOwnerEmail(data, id);
    // Customer ko email
    if (data.email) sendCustomerEmail(data);
    
    return ContentService
      .createTextOutput(JSON.stringify({success: true, id: id}))
      .setMimeType(ContentService.MimeType.JSON);
      
  } catch(err) {
    return ContentService
      .createTextOutput(JSON.stringify({success: false, error: err.message}))
      .setMimeType(ContentService.MimeType.JSON);
  }
}

function getBookings() {
  const sheet  = SpreadsheetApp.getActiveSpreadsheet().getSheetByName(SHEET_NAME);
  const rows   = sheet.getDataRange().getValues();
  const bookings = rows.slice(1).map(r => ({
    id: r[0], submitted: r[1],
    fname: r[2], lname: r[3], email: r[4], phone: r[5],
    date: r[6], time: r[7], guests: r[8],
    seating: r[9], notes: r[10], status: r[11]
  }));
  return ContentService
    .createTextOutput(JSON.stringify({bookings}))
    .setMimeType(ContentService.MimeType.JSON);
}

function updateStatus(id, status) {
  const sheet = SpreadsheetApp.getActiveSpreadsheet().getSheetByName(SHEET_NAME);
  const data  = sheet.getDataRange().getValues();
  for (let i = 1; i < data.length; i++) {
    if (String(data[i][0]) === String(id)) {
      sheet.getRange(i + 1, 12).setValue(status);
      break;
    }
  }
  return ContentService.createTextOutput('OK');
}

function sendOwnerEmail(data, id) {
  const subject = `🔔 Nayi Booking #${id} — ${data.fname} ${data.lname} | ${data.date} ${data.time}`;
  const body = `
Nayi reservation aayi hai!

Naam:    ${data.fname} ${data.lname}
Phone:   ${data.phone}
Email:   ${data.email}
Date:    ${data.date}
Time:    ${data.time}
Guests:  ${data.guests}
Seating: ${data.seating}
Notes:   ${data.notes || 'Koi nahi'}

Admin panel mein dekho: admin.html
  `;
  MailApp.sendEmail(OWNER_EMAIL, subject, body);
}

function sendCustomerEmail(data) {
  const subject = `✅ Booking Received — ${CAFE_NAME}`;
  const body = `
Namaste ${data.fname}!

Aapki reservation request hamein mil gayi hai. Hum jald hi confirm karenge.

Booking Details:
━━━━━━━━━━━━━━━
📅 Date:    ${data.date}
⏰ Time:    ${data.time}
👥 Guests:  ${data.guests}
🪑 Seating: ${data.seating}
━━━━━━━━━━━━━━━

Koi sawaal ho to call karein: +91 84370 43234

Hum aapka intezaar kar rahe hain! 🙏

— Team ${CAFE_NAME}
📍 Coal Depot Complex, Sector 10D, Chandigarh
  `;
  MailApp.sendEmail(data.email, subject, body);
}

Part 3 — config.js Update Karo

6

config.js mein URL paste karo

File config.js kholо aur ye line update karo:

GOOGLE_SCRIPT_URL: 'https://script.google.com/macros/s/YOUR_ID/exec',

Wahan jo URL mili thi Deploy se, woh paste karo.

🎉 Ho gaya! Ab reservation form submit hone par:
✅ Google Sheets mein data save hoga
✅ Owner ko email aayegi
✅ Customer ko confirmation email aayegi
✅ Admin panel mein dikhai dega

Client Change Karna Ho To

Sirf config.js file mein ye update karo:

  • OWNER_EMAIL → Client ki email
  • CAFE_WHATSAPP → Client ka WhatsApp number
  • ADMIN_PASSWORD → Naya password
  • GOOGLE_SCRIPT_URL → Naya Apps Script URL (agar nayi sheet)
💡 Apps Script mein bhi OWNER_EMAIL update karna mat bhoolo — woh alag file hai.
